// === PASTE THIS IN APPS SCRIPT (bound to sheet ID: 1ij3UKjHE-bqJdp7ZK_8lcGp2nkDmHi_icUEy_CKuNQY) ===
function doPost(e) {
  try {
    const sheet = SpreadsheetApp.openById("1ij3UKjHE-bqJdp7ZK_8lcGp2nkDmHi_icUEy_CKuNQY").getSheetByName("GoldData");
    if (!sheet) {
      // create sheet tab if missing
      const ss = SpreadsheetApp.openById("1ij3UKjHE-bqJdp7ZK_8lcGp2nkDmHi_icUEy_CKuNQY");
      const newSheet = ss.insertSheet("GoldData");
      const headers = ["Timestamp","Mine Visited","Location","Coordinates","Size of Mine","Average Grade","No.Samples","ResultsList","Price Asking","Distance Harare","Distance Chinhoyi","Last Inspection","Prev Gold Month","Year","Weekly Prod","Ownership","Depth Belt","Surrounding Activity","Fuel Cost","Belt Size","AvgGrade History Source","Fidelity kgs","Next Review","Comment","ImageUrls"];
      newSheet.getRange(1,1,1,headers.length).setValues([headers]);
      newSheet.getRange("1:1").setFontWeight("bold");
    }
    const data = JSON.parse(e.postData.contents);
    const row = [
      new Date(),
      data.mineName, data.location, data.coords, data.mineSize, data.avgGrade,
      data.samplesCount, data.resultsList, data.priceAsking, data.distHarare,
      data.distChinhoyi, data.lastInspection, data.prevGoldMonth, data.prodYear,
      data.weeklyProd, data.ownership, data.depthBelt, data.surroundingActivity,
      data.fuelCost, data.beltSize, data.gradeHistorySource, data.fidelityKgs,
      data.reviewDate, data.comment, data.imageUrls
    ];
    sheet.appendRow(row);
    return ContentService.createTextOutput(JSON.stringify({status:"success", message:"Record added to Gold sheet"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:"error", message:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
